sh03.sh ：

#!/bin/bash
# Autuor : kiritoXF
# Date : 20180506
# start script
echo "welcome to my world"
website="kirito.space"
echo "my homepage is ${website} have a look!"
read_only="this variable can't be changed"
readonly read_only
echo "btw, what's your name?"
read your_name
greetingStrA="Hello, I know you are \"${your_name}\"! \n"
greetingStrB="welcome to ${website}!"
echo "StrA: ${greetingStrA} \n"
echo "StrB: ${greetingStrB} \n"
echo "strA's length is ${#greetingStrA}"
echo "the substring from 1 to 4 of StrA is ${greetingStrA:1:4}"
echo `expr index "${greetingStrA}" hk`
array=(1,2,3,4)
array2=(
"magic"
"guide"
"the"
"gatherig"
)
array[0]=5
echo ${array[0]}
echo ${array[@]}
length=${#array[@]}
length=${#array[*]}
lengthNum=${#array2[0]}
echo ${length} ${lengthNum}
